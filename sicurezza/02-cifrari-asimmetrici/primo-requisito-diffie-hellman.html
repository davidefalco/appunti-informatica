int expmod(int a, int b, int q){
	// restituisce un valore minore di q
	if(b == 0) return 1;
	if(b % 2 == 0) // se b pari
		return sq(expmod(a, b/2, q)) % q;
	else // b dispari
		return (a * expmod(a, b - 1, q)) % q;
}
